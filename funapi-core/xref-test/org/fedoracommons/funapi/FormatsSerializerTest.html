52  public class FormatsSerializerTest {
53  
54      private XpathEngine engine;
55  
56      @Before
57      public void setUp() {
58          XMLUnit.setIgnoreWhitespace(true);
59          engine = XMLUnit.newXpathEngine();
60          Map<String, String> nsMap = new HashMap<String, String>();
61          NamespaceContext ctx = new SimpleNamespaceContext(nsMap);
62          engine.setNamespaceContext(ctx);
63      }
64  
65      @Test
66      public void testEmptyFormats() throws Exception {
67          UnapiFormats formats = new UnapiFormats(null);
68          FormatsSerializer ser = new FormatsSerializer();
69          Diff diff = new Diff("<formats/>", ser.toString(formats));
70          assertTrue("Expected empty formats element", diff.identical());
71      }
72  
73      @Test
74      public void test2Formats() throws Exception {
75          UnapiFormats formats = new UnapiFormats(null);
76          UnapiFormat oai_dc = new UnapiFormat("oai_dc", "text/xml", "http://www.openarchives.org/OAI/2.0/oai_dc.xsd");
77          formats.addFormat(oai_dc);
78          UnapiFormat mods = new UnapiFormat("mods", "application/xml", "http://www.loc.gov/standards/mods/v3/mods-3-2.xsd");
79          formats.addFormat(mods);
80  
81          Document doc = getDocument(formats);
82          NodeList nodes;
83          Node node;
84          Element e;
85          
86          nodes = engine.getMatchingNodes("/formats/format[@name='oai_dc']", doc);
87          assertEquals(1, nodes.getLength());
88          node = nodes.item(0);
89          assertEquals(Node.ELEMENT_NODE, node.getNodeType());
90          e = (Element) node;
91          assertEquals("text/xml", e.getAttribute("type"));
92          assertEquals("http://www.openarchives.org/OAI/2.0/oai_dc.xsd", e.getAttribute("docs"));
93          
94          nodes = engine.getMatchingNodes("/formats/format[@name='mods']", doc);
95          assertEquals(1, nodes.getLength());
96          node = nodes.item(0);
97          assertEquals(Node.ELEMENT_NODE, node.getNodeType());
98          e = (Element) node;
99          assertEquals("application/xml", e.getAttribute("type"));
100         assertEquals("http://www.loc.gov/standards/mods/v3/mods-3-2.xsd", e.getAttribute("docs"));
101     }
102     
103     private Document getDocument(UnapiFormats formats)
104             throws UnsupportedEncodingException, SAXException, IOException, UnapiException {
105         FormatsSerializer ser = new FormatsSerializer();
106         return XMLUnit.buildControlDocument(ser.toString(formats));
107     }
