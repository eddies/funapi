66  /**
67   * Implementation of ObjectResolver for a Fedora repository.
68   * This implementation will attempt to load a properties file with the name
69   * <code>FedoraResolver.properties</code>.
70   *
71   * Required properties:
72   * <dl>
73   *   <dt>baseURL</dt>
74   *     <dd>The base URL of the Fedora repository, e.g. http://localhost:8080/fedora/</dd>
75   *   <dt>formatsDatastream</dt>
76   *     <dd>The id of a content model object's inline XML datastream which
77   *     describes the unAPI formats available for a Fedora object, e.g.
78   *     UNAPI-FORMATS</dd>
79   * </dl>
80   *
81   * The formatsDatastream must return an XML document that contains a JSON array
82   * describing the available formats. For example:
83   * <pre>&lt;json&gt;
84  [[&quot;info:fedora/&#42;/DC&quot;,&quot;oai_dc&quot;,&quot;text/xml&quot;,&quot;http://www.openarchives.org/OAI/2.0/oai_dc.xsd&quot;],
85   [&quot;info:fedora/&#42;/demo:dc2mods.sdef/transform&quot;,&quot;mods&quot;,&quot;application/xml&quot;,&quot;http://www.loc.gov/standards/mods/&quot;]]
86  &lt;/json&gt;</pre>
87   *
88   * The above is a JSON array of arrays surrounded by &lt;json&gt; tags.
89   * Each four-element inner array represents a single unAPI format.
90   * The first element is the dissemination type URI which will requested object
91   * in the requested format.
92   * The remaining three elements are the unAPI format, type and docs.
93   *
94   * @author Edwin Shin
95   * @since 0.1
96   * @version $Id$
97   */
98  public class FedoraResolver
99          implements ObjectResolver {

108     private final static String HAS_MODEL_URI = "info:fedora/fedora-system:def/model#hasModel";

112     public FedoraResolver() throws UnapiException {
113         InputStream in = getClass().getResourceAsStream("/FedoraResolver.properties");
114         props = new Properties();
115         try {
116             props.load(in);
117         } catch (IOException e) {
118             throw new UnapiException(e.getMessage(), e);
119         }
120 
121         // e.g. http://localhost:8080/fedora
122         String temp = props.getProperty("baseURL");
123         if (!temp.endsWith("/")) {
124             temp += "/";
125         }
126         try {
127             baseURL = new URL(temp);
128         } catch (MalformedURLException e) {
129             throw new UnapiException(e.getMessage(), e);
130         }
131 
132         username = props.getProperty("username");
133         password = props.getProperty("password");
134         formatsDS = props.getProperty("formatsDatastream");
135         defaultFormats = props.getProperty("defaultFormats");
136         FedoraCredentials credentials = new FedoraCredentials(baseURL, username, password);
137         fedoraClient = new FedoraClient(credentials);
138         jsonFactory = new JsonFactory();
139     }

141     public UnapiFormats getFormats() throws UnapiException {
142         UnapiFormats formats;
143         if (defaultFormats == null) {
144             formats = new UnapiFormats(null);
145             UnapiFormat format = new UnapiFormat("oai_dc","text/xml","http://www.openarchives.org/OAI/2.0/oai_dc.xsd");
146             formats.addFormat(format);
147         } else {
148             try {
149                 formats = parseJsonArray(null, defaultFormats);
150             } catch (IOException e) {
151                 throw new UnapiException(e.getMessage(), e);
152             }
153         }
154         return formats;
155     }

181     /**
182      * Returns a resolvable Fedora URL for a PID or info:fedora uri.
183      *
184      * @param id
185      * @return
186      */
187     private String getURL(String id) {
188         String newId = null;
189         if (id.startsWith(FEDORA_URI)) {
190             newId = id.substring(FEDORA_URI.length());
191         } else {
192             newId = id;
193         }
194         return baseURL + "get/" + newId;
195     }

197     /**
198      *
199      * @param id an info:fedora uri (e.g. info:fedora/demo:1)
200      * @return a JSON array of arrays, e.g. [["disstype","format","type","docs"],[...]]
201      * @throws IOException
202      * @throws ServiceException
203      * @throws UnapiException
204      * @throws FedoraClientException
205      */
206     private String getJsonArray(String id) throws IOException, UnapiException, FedoraClientException {
207         String pid = id.substring(FEDORA_URI.length());
208         //FedoraResponse response = getRelationships(pid).subject(id).predicate(HAS_MODEL_URI).format("n-triples").execute(fedoraClient);
209         FedoraResponse response = getRelationships(pid).subject(id).predicate(HAS_MODEL_URI).format("n-triples").execute(fedoraClient);
210 
211         List<String> cmodels = new ArrayList<String>();
212         Model model = ModelFactory.createDefaultModel();
213         model.read(response.getEntityInputStream(), null, FileUtils.langNTriple);
214         StmtIterator sit = model.listStatements();
215         while (sit.hasNext()) {
216             cmodels.add(sit.next().getObject().toString());
217         }
218 
219         DocumentBuilderFactory domFactory = DocumentBuilderFactory.newInstance();
220         DocumentBuilder builder;
221         Document doc = null;
222         try {
223             builder = domFactory.newDocumentBuilder();
224         } catch (ParserConfigurationException e) {
225             throw new UnapiException(e.getMessage(), e);
226         }
227 
228         JsonTypeMapper jtMapper = new JsonTypeMapper();
229         JsonNode rootNode = null;
230         ListDatastreamsResponse ldr;
231         List<DatastreamType> cmodelDatastreams;
232         for (String cmodel : cmodels) {
233             response = null;
234             // check the content model's datastreams for formatsDS
235             ldr = listDatastreams(cmodel).execute(fedoraClient);
236             cmodelDatastreams = ldr.getDatastreams();
237             for (DatastreamType dt : cmodelDatastreams) {
238                 if (dt.getDsid().equals(formatsDS)) {
239                     response = getDatastreamDissemination(cmodel, formatsDS).execute(fedoraClient);
240                     continue;
241                 }
242             }
243 
244             if (response == null) {
245                 continue;
246             }
247 
248             try {
249                 doc = builder.parse(response.getEntityInputStream());
250             } catch (SAXException e) {
251                 throw new UnapiException(e.getMessage(), e);
252             }
253 
254             Element jsonElement = (Element)doc.getElementsByTagName("json").item(0);
255 
256             String json = jsonElement.getTextContent();
257             JsonNode node = jtMapper.read(jsonFactory.createJsonParser(new StringReader(json)));
258             if (rootNode == null) {
259                 rootNode = node;
260             } else {
261                 Iterator<JsonNode> it = node.getElements();
262                 while(it.hasNext()) {
263                     rootNode.appendElement(it.next());
264                 }
265             }
266         }
267 
268         if (rootNode == null) {
269             return null;
270         } else {
271             StringWriter sw = new StringWriter();
272             JsonGenerator gen = jsonFactory.createJsonGenerator(sw);
273             rootNode.writeTo(gen);
274             gen.close();
275             return sw.toString();
276         }
277     }

279     private UnapiFormats parseJsonArray(String id, String jsonContent) throws IOException, UnapiException {
280         UnapiFormats uFormats = new UnapiFormats(id);
281         if (jsonContent == null) {
282             return uFormats;
283         }
284         Set<String> formats = new HashSet<String>();
285 
286         JsonNode rootNode = new JsonTypeMapper().read(jsonFactory.createJsonParser(new StringReader(jsonContent)));
287         Iterator<JsonNode> it = rootNode.getElements();
288         while (it.hasNext()) {
289             JsonNode innerArray = it.next();
290             String format = innerArray.getElementValue(1).getTextValue();
291             String type = innerArray.getElementValue(2).getTextValue();
292             String docs = innerArray.getElementValue(3).getTextValue();
293 
294             if (formats.add(format)) {
295                 UnapiFormat uFormat = new UnapiFormat(format, type, docs);
296                 uFormats.addFormat(uFormat);
297             }
298         }
299         return uFormats;
300     }

302     private String getDisseminationUrl(String id, String format) throws IOException, UnapiException, FedoraClientException {
303         String pid = id.substring(FEDORA_URI.length());
304         String dissType = null;
305 
306         String jsonContent = getJsonArray(id);
307         JsonNode rootNode = new JsonTypeMapper().read(jsonFactory.createJsonParser(new StringReader(jsonContent)));
308 
309         Iterator<JsonNode> it = rootNode.getElements();
310         while (it.hasNext()) {
311             JsonNode innerArray = it.next();
312 
313             String fmt = innerArray.getElementValue(1).getTextValue();
314             if (format.equalsIgnoreCase(fmt)) {
315                 dissType = innerArray.getElementValue(0).getTextValue();
316                 break;
317             }
318         }
319         if (dissType == null) {
320             throw new UnapiException("No dissType for " + id);
321         }
322         // dissType example: info:fedora/*/sdef:1/methodFoo
323         dissType = dissType.replaceAll("\\*", pid);
324         return getURL(dissType);
325     }
